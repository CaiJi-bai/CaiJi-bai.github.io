# 代码执行时间提升技巧  
+ 把执行概率大的代码段放到`if`分支里。

# 模板
1. Union-Find 模板  
```  
class UF {
    public:
        UF(int n) {
            count = n;
            parent = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
            }
        }
        /* 将 p 和 q 连接 */
        void unionTwo(int p, int q) {
            int pRoot = find(p);
            int qRoot = find(q);
            if (pRoot == qRoot)
                return;
            parent[pRoot] = qRoot;
            count--;
        }
        /* 判断 p 和 q 是否连通 */
        bool connected(int p, int q) {
            int pRoot = find(p);
            int qRoot = find(q);
            return pRoot == qRoot;
        }
        /* 返回图中有多少个连通分量 */
        int countN() {
            return count;
        }
    private:
        int find(int x) {
            // 根节点的 parent[x] == x
            while (parent[x] != x) {
                parent[x] = parent[parent[x]];
                x = parent[x];
            }
            return x;
        }

        int count;
        int *parent;
};
```
